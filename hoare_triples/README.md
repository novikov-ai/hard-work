# Триплы Хоара

# Пример 1

~~~go
{ true }
a = 2;
{ a > 1 }
b = 3;
{ a > 1, b > 1 }
c = a + b;
{ c > 1 }
~~~

~~~go
{ true }
a = 1.1;
{ a > 1 }
b = 2.1;
{ a > 1, b > 1 }
c = a + b;
{ c > 1 }
~~~

~~~go
{ true }
a = -3;
{ a < 0 }
b = 6;
{ a < 0, b > 1 - a}
c = a + b;
{ c > 1 }
~~~

~~~go
{ true }
a = 0;
{ a = 0 }
b = 2;
{ a = 0, b > 1}
c = a + b;
{ c > 1 }
~~~

## Пример 2

~~~go
{ a = 0 } 
b = 2 - a;
{ b = 2 } 
c = b * 2; 
{ c = 4 }
d = c + 1;
{ d  =  5 }
~~~
